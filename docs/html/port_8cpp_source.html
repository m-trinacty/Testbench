52 int port::set_port_attribs(int fd, int speed, int parity){
53  struct termios tty;
54  if (tcgetattr (fd, &tty) != 0){
55  syslog(LOG_ERR,"Error %d from tcgetattr ",errno);
56 
57  return EXIT_FAILURE;
58  }
59 
60  cfsetospeed (&tty, speed);
61  cfsetispeed (&tty, speed);
62 
63  tty.c_cflag = (m_port_cfg.c_cflag & ~CSIZE) | CS8; // 8-bit chars
64  // disable IGNBRK for mismatched speed tests; otherwise receive break
65  // as \000 chars
66  tty.c_iflag &= ~IGNBRK; // disable break processing
67  tty.c_lflag = 0; // no signaling chars, no echo,
68  // no canonical processing
69  tty.c_oflag = 0; // no remapping, no delays
70  tty.c_cc[VMIN] = 0; // read doesn't block
71  tty.c_cc[VTIME] = 5; // 0.5 seconds read timeout
72 
73  tty.c_iflag &= ~(IXON | IXOFF | IXANY); // shut off xon/xoff ctrl
74 
75  tty.c_cflag |= (CLOCAL | CREAD);// ignore modem controls,
76  // enable reading
77  tty.c_cflag &= ~(PARENB | PARODD); // shut off parity
78  tty.c_cflag |= parity;
79  tty.c_cflag &= ~CSTOPB;
80  tty.c_cflag &= ~CRTSCTS;
81 
82  if (tcsetattr (fd, TCSANOW, &tty) != 0){
83  syslog(LOG_ERR,"Error %d from tcsetattr ",errno);
84  return EXIT_FAILURE;
85  }
86  return EXIT_SUCCESS;
87 }
